*
    *

    • *

  • something here
  • some other stuff here
}* %{ if (!_body) { throw new play.exceptions.TagInternalException("body cannot be empty for spaceless tag"); } body = _body.toString().trim().replaceAll(">\\s+<", "><").raw() }% ${body}
